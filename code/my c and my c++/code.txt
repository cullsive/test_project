void colorPrint(char *texte, char *couleur) {
    if (strcmp(couleur, "rouge") == 0) {
        printf("\033[0;31m%s\033[0m\n", texte);
    } else if (strcmp(couleur, "vert") == 0) {
        printf("\033[0;32m%s\033[0m\n", texte);
    } else if (strcmp(couleur, "jaune") == 0) {
        printf("\033[0;33m%s\033[0m\n", texte);
    } else if (strcmp(couleur, "bleu") == 0) {
        printf("\033[0;34m%s\033[0m\n", texte);
    } else {
        // Couleur inconnue → texte normal
        printf("%s\n", texte);
    }
}

char* getColorCode(char* couleur) {
    if (strcmp(couleur, "rouge") == 0) return "\033[0;31m";
    if (strcmp(couleur, "vert") == 0) return "\033[0;32m";
    if (strcmp(couleur, "jaune") == 0) return "\033[0;33m";
    if (strcmp(couleur, "bleu") == 0) return "\033[0;34m";
    return "\033[0m"; // couleur par défaut (normal)
}

void printColor(char* texte, char* couleur) {
    printf("%s%s\033[0m\n", getColorCode(couleur), texte);
}

void progress_with_delay(int milliseconds_step) {
    const char *bar = "__________";
printf("\n");
    for (int i = 0; i <= 100; i += 10) {
        struct timespec ts;
        ts.tv_sec = milliseconds_step / 1000;
        ts.tv_nsec = (milliseconds_step % 1000) * 1000000;
        nanosleep(&ts, NULL);

        printf("\r\033[1;32mProgression %-10.*s %d%%\033[0m", i / 10, bar,i);
        fflush(stdout);
    }
    printf("\n");
}

void sleep_ms(int milliseconds) {
    struct timespec ts;
    ts.tv_sec = milliseconds / 1000;
    ts.tv_nsec = (milliseconds % 1000) * 1000000;
    nanosleep(&ts, NULL);
}
